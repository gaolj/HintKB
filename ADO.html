创建mdb数据库文件：

	HRESULT hr0 = S_OK;         
	CString strcdb(CString("Provider=Microsoft.jet.OLEDB.4.0;Data Source =") + dbPath);   
	try   
	{   
		ADOX::_CatalogPtr   m_pCatalog   =   NULL;   
		hr0   =   m_pCatalog.CreateInstance(__uuidof(ADOX::Catalog));   
		if(FAILED(hr0))   
		{      
			_com_issue_error(hr0);   
		}   
    
		else   
		{   
			m_pCatalog->Create(_bstr_t(strcdb));   //Create   MDB   
		}
		return true;
	}       
	catch(_com_error   &e)   
	{		
		CString error = e.Description();
		AfxMessageBox(CString("创建数据库") + dbPath + "失败：" + error);
		return false;
	}
